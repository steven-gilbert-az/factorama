586 
587  int num_passed = 0;
588  int num_failed = 0;
589 
590  for (size_t factor_idx = 0; factor_idx < factors_.size(); ++factor_idx)
591  {

593 
594  // Get analytical jacobians
595  std::vector<Eigen::MatrixXd> J_analytic;
596  factor->compute_jacobians(J_analytic);
597 
598  // Get numerical jacobians
599  std::vector<Eigen::MatrixXd> J_numeric;
600  ComputeNumericalJacobians(*factor, J_numeric);
601 
602  // Compare sizes
603  if (J_analytic.size() != J_numeric.size())
604  {
605  if (verbose)
606  {
607  std::cout << "\n--- Factor " << factor_idx << " (" << factor->name() << ", ID=" << factor->id() << ") ---" << std::endl;
608  std::cout << "FAIL: Jacobian count mismatch - analytical: " << J_analytic.size()
609  << ", numerical: " << J_numeric.size() << std::endl;
610  }
611  num_failed++;
612  continue;
613  }
614 
615  bool factor_passed = true;
616  const auto &variables = factor->variables();
617 
618  for (size_t var_idx = 0; var_idx < J_analytic.size(); ++var_idx)
619  {
620  const auto &var = variables[var_idx];
621  const auto &J_a = J_analytic[var_idx];
622  const auto &J_n = J_numeric[var_idx];
623 
624  // Handle constant variables (empty jacobians)
625  if (var->is_constant())
626  {
627  if (J_a.size() == 0 && J_n.size() == 0)
628  {
629  continue;
630  }
631  else
632  {
633  if (verbose)
634  {
635  if (factor_passed)
636  {
637  std::cout << "\n--- Factor " << factor_idx << " (" << factor->name() << ", ID=" << factor->id() << ") ---" << std::endl;
638  }
639  std::cout << " FAIL: Variable " << var_idx << " (" << var->name() << ", ID=" << var->id()
640  << ") - constant but jacobians not empty" << std::endl;
641  }
642  factor_passed = false;
643  continue;
644  }
645  }
646 
647  // Check dimensions
648  if (J_a.rows() != J_n.rows() || J_a.cols() != J_n.cols())
649  {
650  if (verbose)
651  {
652  if (factor_passed)
653  {
654  std::cout << "\n--- Factor " << factor_idx << " (" << factor->name() << ", ID=" << factor->id() << ") ---" << std::endl;
655  }
656  std::cout << " FAIL: Variable " << var_idx << " (" << var->name() << ", ID=" << var->id()
657  << ") - dimension mismatch" << std::endl;
658  std::cout << " Analytical: " << J_a.rows() << "x" << J_a.cols() << std::endl;
659  std::cout << " Numerical: " << J_n.rows() << "x" << J_n.cols() << std::endl;
660  }
661  factor_passed = false;
662  continue;
663  }
664 
665  // Compare values
666  Eigen::MatrixXd diff = J_n - J_a; // numerical - analytical
667  double max_error = diff.cwiseAbs().maxCoeff();
668  double sum_abs_error = diff.cwiseAbs().sum();
669 
670  if (max_error >= jacobian_tol)
671  {
672  if (verbose)
673  {
