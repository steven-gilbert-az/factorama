219  void SparseOptimizer::single_step_levenberg_marquardt()
220  {
221  if (!graph_) {
222  throw std::runtime_error("[SparseOptimizer] No factor graph set.");
223  }
224 
225  const auto t_start = std::chrono::steady_clock::now();
226 
227  // Step 1: Compute residual and Jacobian
228  const Eigen::VectorXd r = graph_->compute_full_residual_vector();
229  const Eigen::SparseMatrix<double> &J = graph_->compute_sparse_jacobian_matrix();
230 
231  double current_cost = 0.5 * r.squaredNorm();
232 
233  // Initialize damping parameter if this is the first iteration
234  if (current_stats_.current_iteration == 0) {
235  current_stats_.damping_parameter = settings_.initial_lambda;
236  }
237 
238  bool step_accepted = false;
239  Eigen::VectorXd dx;
240  double new_cost = current_cost;
241  int lm_attempts = 0;
242  const int max_lm_attempts = 5;
243 
244  // Levenberg-Marquardt loop: try different damping values until we get improvement
245  while (!step_accepted && lm_attempts < max_lm_attempts) {
246  // Step 2: Form damped normal equations: (J^T J + λI) dx = -J^T r
247  Eigen::SparseMatrix<double> H = J.transpose() * J;
248 
249  // Add damping: H = J^T J + λI
250  for (int i = 0; i < H.outerSize(); ++i) {
251  H.coeffRef(i, i) += current_stats_.damping_parameter;
252  }
253 
254  Eigen::VectorXd b = -J.transpose() * r;
255 
256  // Step 3: Solve damped system
257  sparse_solver_.compute(H);
258  if (sparse_solver_.info() != Eigen::Success) {
259  // If factorization fails, increase damping and try again
260  current_stats_.damping_parameter *= settings_.lambda_up_factor;
261  lm_attempts++;
262  if (settings_.verbose) {
263  std::cout << "[LM] Factorization failed, increasing lambda to " << current_stats_.damping_parameter << "\n";
264  }
265  continue;
266  }
267 
268  dx = sparse_solver_.solve(b);
269  if (sparse_solver_.info() != Eigen::Success) {
270  current_stats_.damping_parameter *= settings_.lambda_up_factor;
271  lm_attempts++;
272  if (settings_.verbose) {
273  std::cout << "[LM] Solve failed, increasing lambda to " << current_stats_.damping_parameter << "\n";
274  }
275  continue;
276  }
277 
278  dx *= settings_.learning_rate;
279 
280  // Step 4: Apply increment to variables (temporarily)
281  std::vector<Eigen::VectorXd> original_values;
282  const auto &variables = graph_->get_all_variables();
283 
284  // Store original values for potential rollback
285  for (const auto &var : variables) {
286  original_values.push_back(var->value());
287  }
288 
289  // Apply increments
290  for (const auto &var : variables) {
291  if (var->is_constant()) {
292  continue;
293  }
294 
295  bool valid = false;
296  VariablePlacement placement = graph_->variable_placement(var->id(), valid);
297  if (!valid) {
298  std::cerr << "[LM] Skipping variable id=" << var->id() << " (no placement found)\n";
299  continue;
300  }
301 
302  const Eigen::VectorXd local_dx = dx.segment(placement.index, placement.dim);
303  var->apply_increment(local_dx);
304  }
305 
306  // Step 5: Evaluate new cost
307  const Eigen::VectorXd r_new = graph_->compute_full_residual_vector();
308  new_cost = 0.5 * r_new.squaredNorm();
309 
310  // Step 6: Check if step should be accepted
311  if (new_cost < current_cost) {
312  // Accept step: decrease damping for next iteration
313  step_accepted = true;
314  current_stats_.damping_parameter *= settings_.lambda_down_factor;
315  current_stats_.damping_parameter = std::max(current_stats_.damping_parameter, 1e-12);
316 
317  if (settings_.verbose) {
318  std::cout << "[LM] Step accepted, cost: " << current_cost << " -> " << new_cost
319  << ", decreasing lambda to " << current_stats_.damping_parameter << "\n";
320  }
321  } else {
322  // Reject step: restore original values and increase damping
323  for (size_t i = 0; i < variables.size(); ++i) {
324  variables[i]->set_value_from_vector(original_values[i]);
325  }
326 
327  current_stats_.damping_parameter *= settings_.lambda_up_factor;
328  current_stats_.damping_parameter = std::min(current_stats_.damping_parameter, settings_.max_lambda);
329  lm_attempts++;
330 
331  if (settings_.verbose) {
332  std::cout << "[LM] Step rejected, cost: " << current_cost << " -> " << new_cost
333  << ", increasing lambda to " << current_stats_.damping_parameter << "\n";
334  }
335 
336  // If lambda becomes too large, we're stuck
337  if (current_stats_.damping_parameter >= settings_.max_lambda) {
338  if (settings_.verbose) {
339  std::cout << "[LM] Lambda too large (" << current_stats_.damping_parameter
340  << " >= " << settings_.max_lambda << "), stopping\n";
341  }
342  break;
343  }
344  }
345  }
346

348  if (!step_accepted) {
349  if (settings_.verbose) {
350  std::cout << "[LM] No acceptable step found after " << max_lm_attempts << " attempts\n";
351  }
352  // Set minimal step for convergence detection
353  dx = Eigen::VectorXd::Zero(dx.size());
354  new_cost = current_cost;
355  }

359  current_stats_.delta_norm = dx.norm();
